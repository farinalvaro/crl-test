## Install

```bash
npm install --save crl-test
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'crl-test'
import 'crl-test/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
